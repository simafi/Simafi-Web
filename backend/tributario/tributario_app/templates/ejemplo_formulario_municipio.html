<script> var municipioCodigo = '{{ empresa }}'; var municipioDescripcion = '{{ municipio_descripcion }}'; // Ejemplo de uso console.log('Código de municipio:', municipioCodigo); console.log('Descripción de municipio:', municipioDescripcion); // Validar que el municipio esté seleccionado if (!municipioCodigo) { alert('Debe seleccionar un municipio'); } </script>
